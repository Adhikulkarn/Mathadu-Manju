STEP 1 — Implement singleton WebSocket

Create a single persistent socket connection.

src/services/voiceSocket.ts

let socket: WebSocket | null = null

export function connectVoice(onAudio:(data:ArrayBuffer)=>void){

if(socket && socket.readyState === WebSocket.OPEN){
return socket
}

socket = new WebSocket("ws://localhost:4000")

socket.binaryType = "arraybuffer"

socket.onopen = ()=>{
console.log("Voice socket connected")
}

socket.onerror = (err)=>{
console.error("WebSocket error",err)
}

socket.onclose = ()=>{
console.log("Voice socket closed")
}

socket.onmessage = (event)=>{
onAudio(event.data)
}

return socket
}

export function sendAudio(data:Blob){

if(socket && socket.readyState === WebSocket.OPEN){
socket.send(data)
}

}

export function disconnectVoice(){

if(socket){
socket.close()
socket = null
}

}
